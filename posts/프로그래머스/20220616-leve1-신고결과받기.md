import (

func solution(id_List []string, report []string, k int) []int {
   //같은 사람이 여러번 신고한 경우 1번으로 본다.
    setReport := set(report)
  
    personReportedAndCount := make(map[string]int)
    reporterAndPersonReported := make(map[string][]string)
    result := make([]int, len(id_List))
    
    for _, value := range setReport {
        //splitValue[0] 신고자 ,splitValue[1] 신고당한사람
        splitValue := strings.Split(value, " ")
        reporter := splitValue[0]
        personReported :=splitValue[1]
        
        //신고당한사람별  count 추가
        personReportedAndCount[personReported] = personReportedAndCount[personReported] + 1
        
        reporterAndPersonReported[reporter] = append(reporterAndPersonReported[reporter] , personReported)
  /*      if count, ok := personReportedAndCount[personReported]; ok {
            personReportedAndCount[personReported] = count + 1
        } else {
            personReportedAndCount[personReported] = 1
        }
        
        if persons, ok := reporterAndPersonReported[reporter]; ok {
            reporterAndPersonReported[reporter] = append(persons, personReported)
        } else {
            reporterAndPersonReported[reporter] = append(persons, personReported)
        }*/
    }
    
    for i, id := range id_List {
        ids := reporterAndPersonReported[id]
        
        for _, v := range ids {
            if count, ok := personReportedAndCount[v]; ok {
                if count >= k {
                    result[i]++
                }
            }
        }
    }
    
    return result
}

func set(list []string) []string {
    //key 하나당 값을 1개  (key 가 중복되는 경우 1번만 들어간다)
    setMap := make(map[string]bool) 
    reports := make([]string, 0)
    for _, value := range list {
        if _, ok := setMap[value]; !ok {
            reports = append(reports, value)
            setMap[value] = true
        }
    }
    return reports
}
